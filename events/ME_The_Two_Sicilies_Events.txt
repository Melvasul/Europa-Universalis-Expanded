#Sack of Spain
country_event = {
	id = ME_The_Two_Sicilies_Events.1

	mean_time_to_happen = {
		months = 1
	}

	trigger = {
		NOT = { has_country_flag = tts_sack_spain }
		OR = {
			tag = NAP
			tag = SIC
			tag = TTS
		}
		OR = {
			AND = {
				war_score_against = {
					who = SPA
					value = 75
				}
				SPA = {
					capital_scope = {
						controlled_by = ROOT
					}
				}
			}
			AND = {
				war_score_against = {
					who = CAS
					value = 75
				}
				CAS = {
					capital_scope = {
						controlled_by = ROOT
					}
				}
			}
			AND = {
				war_score_against = {
					who = ARA
					value = 75
				}
				ARA = {
					capital_scope = {
						controlled_by = ROOT
					}
				}
			}
		}
	}

	immediate = {
		hidden_effect = {
			set_country_flag = tts_sack_spain
		}
	}

	option = {
		name = ME_Naples_Events.16.OPT1
		ai_chance = {
			factor = 100
		}

		add_prestige = 10
		add_years_of_income = 3
		if = {
			limit = { war_with = CAS }
			CAS = {
				add_years_of_income = -3
				capital_scope = {
					add_devastation = 20
					add_base_tax = -3
					add_base_production = -1
					add_base_manpower = -2
				}
			}
		}
		else_if = {
			limit = { war_with = ARA }
			ARA = {
				add_years_of_income = -3
				capital_scope = {
					add_devastation = 20
					add_base_tax = -3
					add_base_production = -1
					add_base_manpower = -2
				}
			}
		}
		else_if = {
			limit = { war_with = SPA }
			SPA = {
				add_years_of_income = -3
				capital_scope = {
					add_devastation = 20
					add_base_tax = -3
					add_base_production = -1
					add_base_manpower = -2
				}
			}
		}
		capital_scope = {
			add_base_tax = 3
			add_base_production = 1
			add_base_manpower = 2
		}
	}
}
