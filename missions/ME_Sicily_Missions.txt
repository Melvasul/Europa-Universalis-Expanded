sicily_mission_1 = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		OR = {
			tag = SIC
			AND = {
				was_tag = SIC
				tag = TTS
			}
		}
		OR = {
			NOT = { has_dlc = "Emperor" }
			AND = {
				has_dlc = "Emperor"
				has_country_flag = use_missions_expanded_missions_instead_of_paradox
			}
		}
	}
	has_country_shield = yes

	sic_holy_see = {
		icon = answer_the_roman_question
		required_missions = { }
		position = 1
		trigger = {
			diplomatic_reputation = 2
			if = {
				limit = { exists = PAP }
				PAP = {
					has_opinion = {
						who = ROOT
						value = 100
					}
				}
			}			
		}
		effect = {
			random_owned_province = {
				limit = { has_cardinal = no }
				add_cardinal = yes
			}
			add_papal_influence = 25
		}
	}

	sic_seeking_papal_support = {
		icon = defender_of_the_faith
		required_missions = {
			sic_holy_see
			sic_gain_independence
		}
		position = 2
		trigger = {
			OR = {
				OR = {
					alliance_with = PAP
					AND = {
						has_opinion = {
							who = PAP
							value = 150
						}
						PAP = {
							has_opinion = {
								who = ROOT
								value = 150
							}
						}
					}
				}
				OR = {
					is_rival = PAP
					AND = {
						NOT = {
							has_opinion = {
								who = PAP
								value = -100
							}
						}
						NOT = {
							PAP = {
								has_opinion = {
									who = ROOT
									value = -100
								}
							}
						}
					}
				}
			}
		}
		effect = {
			if = {
				limit = {
					OR = {
						alliance_with = PAP
						AND = {
							has_opinion = {
								who = PAP
								value = 150
							}
							PAP = {
								has_opinion = {
									who = ROOT
									value = 150
								}
							}
						}
					}
				}
				add_country_modifier = {
					name = sic_papal_delegation
					duration = 9125
				}
			}
			else = {
				125 = {
					if = {
						limit = { has_cardinal = no }
						add_cardinal = yes
					}
					else_if = {
						limit = {
							has_cardinal = yes
							has_building = temple
						}
						remove_building = temple
						add_building_construction = {
							building = cathedral
							speed = 0.75
							cost = 0
						}
					}
					else_if = {
						limit = { has_cardinal = yes }
						add_building_construction = {
							building = temple
							speed = 0.75
							cost = 0
						}
					}
					else = {
						add_base_tax = 2
					}
				}
				add_country_modifier = {
					name = sic_anti_pope_approval
					duration = -1
				}
			}
		}
	}
